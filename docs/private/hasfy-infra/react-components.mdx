## 🎨 Composant principal : AgentDownloadInterface

```tsx
// components/agent-install/AgentDownloadInterface.tsx
'use client'

import { useState, useEffect } from 'react'
import { useSupabaseClient, useUser } from '@supabase/auth-helpers-react'
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/Tabs'
import { Button } from '@/components/ui/Button'
import { CopyButton } from '@/components/ui/CopyButton'
import { Check, Download, Terminal, Laptop, Server } from 'lucide-react'

type OSType = 'windows' | 'linux' | 'macos'
type InstallMode = 'manual' | 'one-click'

interface AgentDownloadInterfaceProps {
  equipmentId?: string
  equipmentName?: string
}

export default function AgentDownloadInterface({ 
  equipmentId, 
  equipmentName 
}: AgentDownloadInterfaceProps) {
  const user = useUser()
  const supabase = useSupabaseClient()
  
  const [enrollmentToken, setEnrollmentToken] = useState<string>('')
  const [fleetUrl, setFleetUrl] = useState<string>('')
  const [selectedOS, setSelectedOS] = useState<OSType>('windows')
  const [installMode, setInstallMode] = useState<InstallMode>('one-click')
  const [isAgentLinked, setIsAgentLinked] = useState(false)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    async function fetchEnrollmentToken() {
      setLoading(true)
      try {
        const { data, error } = await supabase
          .from('enrollment_tokens')
          .select('token, fleet_url')
          .eq('client_id', user?.id)
          .eq('is_active', true)
          .single()

        if (error) throw error

        if (data) {
          setEnrollmentToken(data.token)
          setFleetUrl(data.fleet_url)
        }
      } catch (error) {
        console.error('Error fetching enrollment token:', error)
      } finally {
        setLoading(false)
      }
    }

    if (user?.id) {
      fetchEnrollmentToken()
    }
  }, [user, supabase])

  const getAgentVersion = () => '8.17.2'

  const getDownloadURL = (os: OSType) => {
    const version = getAgentVersion()
    const urls = {
      windows: `https://artifacts.elastic.co/downloads/beats/elastic-agent/elastic-agent-${version}-windows-x86_64.zip`,
      linux: `https://artifacts.elastic.co/downloads/beats/elastic-agent/elastic-agent-${version}-linux-x86_64.tar.gz`,
      macos: `https://artifacts.elastic.co/downloads/beats/elastic-agent/elastic-agent-${version}-darwin-x86_64.tar.gz`
    }
    return urls[os]
  }

  const getInstallCommand = (os: OSType) => {
    const tags = equipmentId 
      ? `equipment_id:${equipmentId},user_id:${user?.id}`
      : `user_id:${user?.id}`
    
    const version = getAgentVersion()

    const commands = {
      windows: `$ProgressPreference = 'SilentlyContinue'
Invoke-WebRequest -Uri ${getDownloadURL('windows')} -OutFile elastic-agent-${version}-windows-x86_64.zip 
Expand-Archive .\\elastic-agent-${version}-windows-x86_64.zip -DestinationPath .
cd elastic-agent-${version}-windows-x86_64
.\\elastic-agent.exe install --url=${fleetUrl}:443 --enrollment-token=${enrollmentToken} --tag "${tags}"`,
      
      linux: `curl -L -O ${getDownloadURL('linux')}
tar xzvf elastic-agent-${version}-linux-x86_64.tar.gz
cd elastic-agent-${version}-linux-x86_64
sudo ./elastic-agent install --url=${fleetUrl}:443 --enrollment-token=${enrollmentToken} --tag "${tags}"`,
      
      macos: `curl -L -O ${getDownloadURL('macos')}
tar xzvf elastic-agent-${version}-darwin-x86_64.tar.gz
cd elastic-agent-${version}-darwin-x86_64
sudo ./elastic-agent install --url=${fleetUrl}:443 --enrollment-token=${enrollmentToken} --tag "${tags}"`
    }

    return commands[os]
  }

  const handleLinkAgent = async () => {
    if (!equipmentId) {
      alert('Equipment ID is required')
      return
    }

    try {
      const response = await fetch('/api/equipment/link-agent', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          equipment_id: equipmentId,
          client_id: user?.id
        })
      })

      if (response.ok) {
        setIsAgentLinked(true)
        alert('Agent successfully linked to equipment!')
      } else {
        const error = await response.json()
        alert(`Error linking agent: ${error.message}`)
      }
    } catch (error) {
      console.error('Error linking agent:', error)
      alert('Failed to link agent')
    }
  }

  if (loading) {
    return (
      <div className="flex items-center justify-center min-h-[400px]">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div>
      </div>
    )
  }

  return (
    <div className="max-w-5xl mx-auto p-6 space-y-8">
      {/* Header */}
      <div className="space-y-2">
        <h1 className="text-3xl font-bold text-gray-900">
          Install Hasfy Agent
        </h1>
        <p className="text-gray-600">
          Follow the simple steps below to install the Hasfy monitoring agent on your equipment. 
          The agent securely collects performance and health metrics to help you monitor your system.
        </p>
        {equipmentName && (
          <div className="inline-flex items-center gap-2 px-3 py-1 bg-blue-50 text-blue-700 rounded-full text-sm">
            <Server className="w-4 h-4" />
            <span>Installing for: <strong>{equipmentName}</strong></span>
          </div>
        )}
      </div>

      {/* Installation Mode Tabs */}
      <Tabs value={installMode} onValueChange={(v) => setInstallMode(v as InstallMode)}>
        <TabsList className="grid w-full grid-cols-2">
          <TabsTrigger value="one-click">
            One-Click Installation (Recommended)
          </TabsTrigger>
          <TabsTrigger value="manual">
            Manual Installation
          </TabsTrigger>
        </TabsList>

        {/* One-Click Installation */}
        <TabsContent value="one-click" className="space-y-6">
          <div className="bg-gradient-to-br from-blue-50 to-indigo-50 p-8 rounded-xl border border-blue-200">
            <h2 className="text-xl font-semibold mb-4 text-gray-900">
              One-Click Installation
            </h2>
            <p className="text-gray-600 mb-6">
              Simply download and run the Hasfy Agent installer for your operating system. 
              The installer handles all configuration automatically.
            </p>

            {/* OS Selection */}
            <div className="flex gap-4 mb-6">
              <button
                onClick={() => setSelectedOS('windows')}
                className={`flex-1 p-4 rounded-lg border-2 transition-all ${
                  selectedOS === 'windows'
                    ? 'border-blue-600 bg-blue-50'
                    : 'border-gray-200 bg-white hover:border-blue-300'
                }`}
              >
                <Laptop className="w-8 h-8 mx-auto mb-2" />
                <div className="font-semibold">Windows</div>
              </button>
              <button
                onClick={() => setSelectedOS('linux')}
                className={`flex-1 p-4 rounded-lg border-2 transition-all ${
                  selectedOS === 'linux'
                    ? 'border-blue-600 bg-blue-50'
                    : 'border-gray-200 bg-white hover:border-blue-300'
                }`}
              >
                <Terminal className="w-8 h-8 mx-auto mb-2" />
                <div className="font-semibold">Linux</div>
              </button>
              <button
                onClick={() => setSelectedOS('macos')}
                className={`flex-1 p-4 rounded-lg border-2 transition-all ${
                  selectedOS === 'macos'
                    ? 'border-blue-600 bg-blue-50'
                    : 'border-gray-200 bg-white hover:border-blue-300'
                }`}
              >
                <Laptop className="w-8 h-8 mx-auto mb-2" />
                <div className="font-semibold">macOS</div>
              </button>
            </div>

            {/* Download Button */}
            <div className="text-center">
              <a
                href={getDownloadURL(selectedOS)}
                download
                className="inline-flex items-center gap-2 px-8 py-4 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-colors font-semibold text-lg"
              >
                <Download className="w-5 h-5" />
                Download for {selectedOS.charAt(0).toUpperCase() + selectedOS.slice(1)}
              </a>
              <p className="mt-4 text-sm text-gray-500">
                Version {getAgentVersion()} • {
                  selectedOS === 'windows' ? '.msi installer' : 
                  selectedOS === 'linux' ? '.tar.gz package' : 
                  '.tar.gz package'
                }
              </p>
            </div>

            {/* Installation Steps */}
            <div className="mt-8 space-y-4">
              <div className="flex items-start gap-3">
                <div className="flex-shrink-0 w-8 h-8 rounded-full bg-blue-600 text-white flex items-center justify-center font-semibold">
                  1
                </div>
                <div>
                  <h3 className="font-semibold text-gray-900">Download the installer</h3>
                  <p className="text-sm text-gray-600">Click the download button above</p>
                </div>
              </div>
              <div className="flex items-start gap-3">
                <div className="flex-shrink-0 w-8 h-8 rounded-full bg-blue-600 text-white flex items-center justify-center font-semibold">
                  2
                </div>
                <div>
                  <h3 className="font-semibold text-gray-900">Run the installer</h3>
                  <p className="text-sm text-gray-600">
                    {selectedOS === 'windows' && 'Double-click the .msi file and approve administrator access'}
                    {selectedOS === 'linux' && 'Extract and run the installation script with sudo'}
                    {selectedOS === 'macos' && 'Extract and run the installation script with sudo'}
                  </p>
                </div>
              </div>
              <div className="flex items-start gap-3">
                <div className="flex-shrink-0 w-8 h-8 rounded-full bg-blue-600 text-white flex items-center justify-center font-semibold">
                  3
                </div>
                <div>
                  <h3 className="font-semibold text-gray-900">Installation completes automatically</h3>
                  <p className="text-sm text-gray-600">The agent will start monitoring immediately</p>
                </div>
              </div>
            </div>
          </div>
        </TabsContent>

        {/* Manual Installation */}
        <TabsContent value="manual" className="space-y-6">
          {/* OS Tabs */}
          <Tabs value={selectedOS} onValueChange={(v) => setSelectedOS(v as OSType)}>
            <TabsList className="grid w-full grid-cols-3">
              <TabsTrigger value="windows">Windows</TabsTrigger>
              <TabsTrigger value="linux">Linux</TabsTrigger>
              <TabsTrigger value="macos">macOS</TabsTrigger>
            </TabsList>

            {/* Windows */}
            <TabsContent value="windows" className="space-y-6">
              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">1. Download the Windows Package</h3>
                <Button asChild>
                  <a href={getDownloadURL('windows')} download>
                    <Download className="w-4 h-4 mr-2" />
                    Download Windows Installer
                  </a>
                </Button>
              </div>

              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">2. Run Installation Command</h3>
                <p className="text-sm text-gray-600">
                  Open PowerShell as Administrator and run:
                </p>
                <div className="relative">
                  <pre className="bg-gray-900 text-white p-4 rounded-lg overflow-x-auto text-sm">
                    <code>{getInstallCommand('windows')}</code>
                  </pre>
                  <div className="absolute top-2 right-2">
                    <CopyButton text={getInstallCommand('windows')} />
                  </div>
                </div>
              </div>
            </TabsContent>

            {/* Linux */}
            <TabsContent value="linux" className="space-y-6">
              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">1. Download the Linux Package</h3>
                <Button asChild>
                  <a href={getDownloadURL('linux')} download>
                    <Download className="w-4 h-4 mr-2" />
                    Download Linux Package
                  </a>
                </Button>
              </div>

              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">2. Run Installation Command</h3>
                <p className="text-sm text-gray-600">
                  Open a terminal and run:
                </p>
                <div className="relative">
                  <pre className="bg-gray-900 text-white p-4 rounded-lg overflow-x-auto text-sm">
                    <code>{getInstallCommand('linux')}</code>
                  </pre>
                  <div className="absolute top-2 right-2">
                    <CopyButton text={getInstallCommand('linux')} />
                  </div>
                </div>
              </div>
            </TabsContent>

            {/* macOS */}
            <TabsContent value="macos" className="space-y-6">
              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">1. Download the macOS Package</h3>
                <Button asChild>
                  <a href={getDownloadURL('macos')} download>
                    <Download className="w-4 h-4 mr-2" />
                    Download macOS Package
                  </a>
                </Button>
              </div>

              <div className="bg-gray-50 p-6 rounded-lg space-y-4">
                <h3 className="text-lg font-semibold">2. Run Installation Command</h3>
                <p className="text-sm text-gray-600">
                  Open Terminal and run:
                </p>
                <div className="relative">
                  <pre className="bg-gray-900 text-white p-4 rounded-lg overflow-x-auto text-sm">
                    <code>{getInstallCommand('macos')}</code>
                  </pre>
                  <div className="absolute top-2 right-2">
                    <CopyButton text={getInstallCommand('macos')} />
                  </div>
                </div>
              </div>
            </TabsContent>
          </Tabs>
        </TabsContent>
      </Tabs>

      {/* Link Agent Section */}
      {equipmentId && (
        <div className="bg-green-50 p-6 rounded-lg border border-green-200">
          <h3 className="text-lg font-semibold text-gray-900 mb-2">
            3. Link Agent to Equipment
          </h3>
          <p className="text-gray-600 mb-4">
            After installing the agent, click the button below to automatically link it to this equipment.
          </p>
          
          {isAgentLinked ? (
            <div className="flex items-center gap-2 text-green-700">
              <Check className="w-5 h-5" />
              <span className="font-semibold">Agent successfully linked!</span>
            </div>
          ) : (
            <Button 
              onClick={handleLinkAgent}
              className="bg-green-600 hover:bg-green-700"
            >
              Link Agent to Equipment
            </Button>
          )}
        </div>
      )}

      {/* Help Section */}
      <div className="bg-blue-50 p-6 rounded-lg border border-blue-200">
        <h3 className="text-lg font-semibold text-gray-900 mb-2">
          Need Help?
        </h3>
        <p className="text-gray-600 mb-4">
          If you encounter any issues during installation, please check our documentation or contact support.
        </p>
        <div className="flex gap-4">
          <Button variant="outline">
            View Documentation
          </Button>
          <Button variant="outline">
            Contact Support
          </Button>
        </div>
      </div>
    </div>
  )
}
```

```tsx
// components/ui/Button.tsx
import { ButtonHTMLAttributes, forwardRef } from 'react'
import { cn } from '@/lib/utils'

interface ButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'default' | 'outline' | 'ghost'
  size?: 'sm' | 'md' | 'lg'
  asChild?: boolean
}

const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant = 'default', size = 'md', asChild, children, ...props }, ref) => {
    const Comp = asChild ? 'span' : 'button'
    
    return (
      <Comp
        className={cn(
          'inline-flex items-center justify-center rounded-lg font-medium transition-colors',
          'focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-offset-2',
          'disabled:pointer-events-none disabled:opacity-50',
          {
            'bg-blue-600 text-white hover:bg-blue-700': variant === 'default',
            'border-2 border-gray-300 bg-white hover:bg-gray-50': variant === 'outline',
            'hover:bg-gray-100': variant === 'ghost',
            'px-3 py-1.5 text-sm': size === 'sm',
            'px-4 py-2 text-base': size === 'md',
            'px-6 py-3 text-lg': size === 'lg',
          },
          className
        )}
        ref={ref}
        {...props}
      >
        {children}
      </Comp>
    )
  }
)

Button.displayName = 'Button'

export { Button }
```

### CopyButton Component

```tsx
// components/ui/CopyButton.tsx
'use client'

import { useState } from 'react'
import { Check, Copy } from 'lucide-react'
import { Button } from './Button'

interface CopyButtonProps {
  text: string
}

export function CopyButton({ text }: CopyButtonProps) {
  const [copied, setCopied] = useState(false)

  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(text)
      setCopied(true)
      setTimeout(() => setCopied(false), 2000)
    } catch (error) {
      console.error('Failed to copy:', error)
    }
  }

  return (
    <Button
      size="sm"
      variant="outline"
      onClick={handleCopy}
      className="bg-white"
    >
      {copied ? (
        <>
          <Check className="w-4 h-4 mr-1 text-green-600" />
          Copied!
        </>
      ) : (
        <>
          <Copy className="w-4 h-4 mr-1" />
          Copy
        </>
      )}
    </Button>
  )
}
```

### Tabs Component

```tsx
// components/ui/Tabs.tsx
'use client'

import { createContext, useContext, useState, ReactNode } from 'react'
import { cn } from '@/lib/utils'

interface TabsContextValue {
  value: string
  onValueChange: (value: string) => void
}

const TabsContext = createContext<TabsContextValue | undefined>(undefined)

export function Tabs({ 
  value, 
  onValueChange, 
  children, 
  className 
}: { 
  value: string
  onValueChange: (value: string) => void
  children: ReactNode
  className?: string
}) {
  return (
    <TabsContext.Provider value={{ value, onValueChange }}>
      <div className={className}>{children}</div>
    </TabsContext.Provider>
  )
}

export function TabsList({ children, className }: { children: ReactNode; className?: string }) {
  return (
    <div className={cn('inline-flex h-10 items-center justify-center rounded-lg bg-gray-100 p-1', className)}>
      {children}
    </div>
  )
}

export function TabsTrigger({ value, children }: { value: string; children: ReactNode }) {
  const context = useContext(TabsContext)
  if (!context) throw new Error('TabsTrigger must be used within Tabs')

  const isActive = context.value === value

  return (
    <button
      onClick={() => context.onValueChange(value)}
      className={cn(
        'inline-flex items-center justify-center whitespace-nowrap rounded-md px-3 py-1.5',
        'text-sm font-medium transition-all',
        'focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-offset-2',
        isActive
          ? 'bg-white text-gray-900 shadow-sm'
          : 'text-gray-600 hover:text-gray-900'
      )}
    >
      {children}
    </button>
  )
}

export function TabsContent({ value, children, className }: { value: string; children: ReactNode; className?: string }) {
  const context = useContext(TabsContext)
  if (!context) throw new Error('TabsContent must be used within Tabs')

  if (context.value !== value) return null

  return <div className={cn('mt-4', className)}>{children}</div>
}
```
